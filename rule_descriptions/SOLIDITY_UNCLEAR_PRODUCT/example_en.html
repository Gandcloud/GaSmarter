In the following contract, this is an example of definitioned variable: 


    pragma solidity 0.4.24;

    contract A {
        function deadcode(uint x) public returns(uint){
            if(x<=0){
                if(x<1){
                    return x;
                }
            }    
        }
    }

Better Alternative: 

    
        pragma solidity 0.4.24;
    
        contract A {
            function deadcode(uint x) public returns(uint){
                if(x<=0){
                    return x;
                }        
            }
        }
