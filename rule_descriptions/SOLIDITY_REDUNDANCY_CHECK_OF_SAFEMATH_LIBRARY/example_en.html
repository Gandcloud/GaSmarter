In the following contract, this is an example of definitioned variable: 


    pragma solidity 0.4.24;

    
    import "./safemath.sol";

    contract A {
        require(balance >= amount);
        balance = balance.sub(amount);
      }

Better Alternative: 

    
        pragma solidity 0.4.24;
    
        import "./safemath.sol";

        contract A {
            balance = balance.sub(amount);
          }
