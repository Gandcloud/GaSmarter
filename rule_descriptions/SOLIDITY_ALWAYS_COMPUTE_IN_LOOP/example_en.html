In the following contract, this is an example of definitioned variable: 


    pragma solidity 0.4.24;

    contract A {
    
        uint a = 4;
        uint b = 5;
        function repeatedComputations(uint x) public returns(uint) {
            uint sum = 0;
            for(uint i = 0; i <= x; i++) {
                sum = sum + a * b;
            }
        }
    }

Better Alternative: 

    
        pragma solidity 0.4.24;
    
        contract A {
    
            uint a = 4;
            uint b = 5;
            uint c = a * b;
            function repeatedComputations(uint x) public returns(uint) {
                uint sum = 0;
                for(uint i = 0; i <= x; i++) {
                    sum = sum + c;
                }
            }
        }
